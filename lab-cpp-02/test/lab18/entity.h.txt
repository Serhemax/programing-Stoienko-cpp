#ifndef ENTITY_H
#define ENTITY_H
#include <stdio.h>
#include <stdlib.h>
#include <string>
#include <iostream>
#include <sstream>
using namespace std;

class Bird {
private:
        bool is_ringed; // Чи окольцьована птаха (так, ні)
        string type;    // Назва виду
        int age;       // Вік птаха, місяців
        
        int area;     // Площа у кв. см
        int height;   // Висота у см
        int feeders;  // Кількість годівниць
        bool has_nest; // Наявність гнізда (так, ні)
               // Тип домівки птаха
        string gender;
public:

    Bird();

    Bird(bool is_ringed_in, string type_in, int age_in, int area_in, int height_in,
        int feeders_in, bool has_nest_int, string gender);

    Bird(const Bird& copy);

    static Bird createBirdFromInputString(const string& line);

    const bool getIsringed();
    void setIsringed(bool isringed);

    const string getType();
    void setType(string type);

    const int getAge();
    void setAge(int age);

    const int getArea();
    void setArea(int area);

    const int getHeight();
    void setHeight(int height);

    const int getFeeders();
    void setFeeders(int feeders);

    const bool getHasnest();
    void setHasnest(bool hasnest);

    const string getGender();
    void setGender(string gender);

    /// Функція для виведення масиву птахів на екран
    string toString();

    ~Bird() {};
};
#endif
